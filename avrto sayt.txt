    <script>
        const tg = window.Telegram.WebApp;
        if (tg) tg.expand();

        const imageInput = document.getElementById('imageInput');
        const canvas = document.getElementById('canvas');
        const preview = document.getElementById('preview');
        let finalImageData = null;

        imageInput.addEventListener('change', function(e) {
            const file = e.target.files[0];
            if (!file) return;

            const reader = new FileReader();
            reader.onload = function(event) {
                const img = new Image();
                img.onload = function() {
                    const ctx = canvas.getContext('2d');
                    const border = 30;
                    
                    canvas.width = img.width + (border * 2);
                    canvas.height = img.height + (border * 2);
                    
                    ctx.fillStyle = "black";
                    ctx.fillRect(0, 0, canvas.width, canvas.height);
                    
                    ctx.filter = "contrast(1.2) saturate(1.1)";
                    ctx.drawImage(img, border, border);
                    
                    finalImageData = canvas.toDataURL("image/jpeg", 0.9);
                    preview.src = finalImageData;
                    preview.style.display = "block";
                }
                img.src = event.target.result;
            }
            reader.readAsDataURL(file);
        });

        document.getElementById('sendBtn').addEventListener('click', function() {
            const car = document.getElementById('car').value;
            const part = document.getElementById('part').value;

            if (!finalImageData) {
                alert("Iltimos, rasm yuklang!");
                return;
            }

            const data = {
                car: car,
                part: part,
                image: finalImageData
            };

            if (tg) {
                tg.sendData(JSON.stringify(data));
                tg.close();
            } else {
                alert("Bu sayt faqat Telegram bot ichida ishlaydi!");
            }
        });
    </script>
